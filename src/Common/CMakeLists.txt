set(NAME Common)

set(LOCAL_HEADER_LIST)
actualize_header_list(HEADER_LIST LOCAL_HEADER_LIST)
list(FILTER LOCAL_HEADER_LIST
    INCLUDE
    REGEX "${NAME}"
)
set(HEADER_LIST ${LOCAL_HEADER_LIST})
unset(LOCAL_HEADER_LIST)
message("${NAME} : HEADER_LIST = ${HEADER_LIST}")

set(MODULE_WITH_FILES FALSE)
set(MODULES FALSE)

set_object_module(
    ${NAME}
    ${MODULE_WITH_FILES}
    ${MODULES}
    LIST_OF_MODULES
)
set(${NAME}_MODULES
    ${LIST_OF_MODULES}
    PARENT_SCOPE
)

# Tests
set(ENABLE_TESTS ${CMAKE_ENABLE_TESTS})
add_tests(${ENABLE_TESTS})

set(${NAME}_TEST_MODULE
    ${${NAME}Tests_TEST_MODULE}
    PARENT_SCOPE
)
