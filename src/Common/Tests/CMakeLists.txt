cmake_minimum_required(VERSION 3.14)
set(NAME Common.Tests)
set(SOURCE_LIST_FILE ${CMAKE_CURRENT_SOURCE_DIR}/Sources.cmake)
set(HEADER_LIST_FILE ${CMAKE_CURRENT_SOURCE_DIR}/Headers.cmake)
set(DEPENDENCY_HEADER_LIST ${FLAME_NAMESPACE}::Common::Headers)

compile_library(
	${FLAME_DEBUG}

	${FLAME_STATIC}
	${FLAME_SHARED}

	LIBRARY_NAME              ${NAME}

	LIBRARY_OBJECT_ALIAS_NAME ${FLAME_NAMESPACE}::Common::Tests::Object
	LIBRARY_STATIC_ALIAS_NAME ${FLAME_NAMESPACE}::Common::Tests::Static
	LIBRARY_SHARED_ALIAS_NAME ${FLAME_NAMESPACE}::Common::Tests::Shared

	SOURCE_LIST_FILE          ${SOURCE_LIST_FILE}
	HEADER_LIST_FILE          ${HEADER_LIST_FILE}
	INCLUDE_PATH              "${FLAME_INCLUDE_PATH};${FLAME_INCLUDE_TEST_PATH}"

	DEPENDENCY_HEADER_LIST    "${DEPENDENCY_HEADER_LIST}"
)
