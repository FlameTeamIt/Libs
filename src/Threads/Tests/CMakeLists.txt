cmake_minimum_required(VERSION 3.14)
set(NAME Threads.Tests)
set(SOURCE_LIST_FILE       ${CMAKE_CURRENT_SOURCE_DIR}/Sources.cmake)
set(HEADER_LIST_FILE       ${CMAKE_CURRENT_SOURCE_DIR}/Headers.cmake)
set(DEPENDENCY_HEADER_LIST ${FLAME_NAMESPACE}::Threads::Headers)
set(DEPENDENCY_STATIC_LIST ${FLAME_NAMESPACE}::Threads::Static)
set(DEPENDENCY_SHARED_LIST ${FLAME_NAMESPACE}::Threads::Shared)
compile_library(
	${FLAME_DEBUG}

	${FLAME_STATIC}
	${FLAME_SHARED}

	LIBRARY_NAME              ${NAME}

	LIBRARY_OBJECT_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::Tests::Object
	LIBRARY_STATIC_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::Tests::Static
	LIBRARY_SHARED_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::Tests::Shared

	SOURCE_LIST_FILE          ${SOURCE_LIST_FILE}
	HEADER_LIST_FILE          ${HEADER_LIST_FILE}
	INCLUDE_PATH              "${FLAME_INCLUDE_PATH};${FLAME_INCLUDE_TEST_PATH}"

	DEPENDENCY_HEADER_LIST    "${DEPENDENCY_HEADER_LIST}"
	DEPENDENCY_STATIC_LIST    "${DEPENDENCY_STATIC_LIST}"
	DEPENDENCY_SHARED_LIST    "${DEPENDENCY_SHARED_LIST}"
)
