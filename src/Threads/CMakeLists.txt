cmake_minimum_required(VERSION 3.14)
set(NAME Threads)
set(DEPENDENCY_HEADER_LIST
	${FLAME_NAMESPACE}::Threads::Headers)

set(DEPENDENCY_OBJECT_LIST
	${FLAME_NAMESPACE}::Threads::${FLAME_PLATFORM}::Object)
set(DEPENDENCY_STATIC_LIST
	${FLAME_NAMESPACE}::Threads::${FLAME_PLATFORM}::Static)
set(DEPENDENCY_SHARED_LIST
	${FLAME_NAMESPACE}::Threads::${FLAME_PLATFORM}::Shared)

set(SOURCE_LIST_FILE ${CMAKE_CURRENT_SOURCE_DIR}/Sources.cmake)

compile_library(
	${FLAME_CMAKE_DEBUG} ${FLAME_STATIC} ${FLAME_SHARED}

	LIBRARY_NAME              ${NAME}

	LIBRARY_OBJECT_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::Object
	LIBRARY_STATIC_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::Static
	LIBRARY_SHARED_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::Shared

	SOURCE_LIST_FILE          ${SOURCE_LIST_FILE}
	INCLUDE_PATH              ${FLAME_INCLUDE_PATH}

	DEPENDENCY_HEADER_LIST    "${DEPENDENCY_HEADER_LIST}"
	DEPENDENCY_OBJECT_LIST    "${DEPENDENCY_OBJECT_LIST}"
	DEPENDENCY_STATIC_LIST    "${DEPENDENCY_STATIC_LIST}"
	DEPENDENCY_SHARED_LIST    "${DEPENDENCY_SHARED_LIST}"
)
