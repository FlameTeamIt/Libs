cmake_minimum_required(VERSION 3.14)

set(NAME                  "Threads${FLAME_NAME_SEPARATOR}Posix")
set(DEPENDENCY_HEADER_LIST ${FLAME_NAMESPACE}::Threads::Headers)
set(DEPENDENCY_STATIC_LIST ${FLAME_DEPENDENCY_SHARED_LIBRARIES})
set(DEPENDENCY_SHARED_LIST ${FLAME_DEPENDENCY_SHARED_LIBRARIES})
set(SOURCE_LIST_FILE       ${CMAKE_CURRENT_SOURCE_DIR}/Sources.cmake)

if ("${FLAME_PLATFORM}" STREQUAL "Posix")
	compile_library(
		${FLAME_DEBUG}

		${FLAME_STATIC}
		${FLAME_SHARED}

		LIBRARY_NAME              ${NAME}

		LIBRARY_OBJECT_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::${FLAME_PLATFORM}::Object
		LIBRARY_STATIC_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::${FLAME_PLATFORM}::Static
		LIBRARY_SHARED_ALIAS_NAME ${FLAME_NAMESPACE}::Threads::${FLAME_PLATFORM}::Shared

		SOURCE_LIST_FILE          ${SOURCE_LIST_FILE}
		INCLUDE_PATH              ${FLAME_INCLUDE_PATH}

		DEPENDENCY_HEADER_LIST    "${DEPENDENCY_HEADER_LIST}"
		DEPENDENCY_STATIC_LIST    "${DEPENDENCY_STATIC_LIST}"
		DEPENDENCY_SHARED_LIST    "${DEPENDENCY_SHARED_LIST}"
	)
endif()
