cmake_minimum_required(VERSION 3.14)
set(NAME                   Streams)
set(DEPENDENCY_HEADER_LIST ${FLAME_NAMESPACE}::Streams::Headers)
set(SOURCE_LIST_FILE       ${CMAKE_CURRENT_SOURCE_DIR}/Sources.cmake)
set(HEADER_LIST_FILE       ${CMAKE_CURRENT_SOURCE_DIR}/Headers.cmake)

set(DEPENDENCY_SHARED_LIST ${FLAME_NAMESPACE}::Threads::Shared)
set(DEPENDENCY_OBJECT_LIST ${FLAME_NAMESPACE}::Streams::${FLAME_PLATFORM}::Object)

compile_library(
	${FLAME_DEBUG}

	${FLAME_STATIC}
	${FLAME_SHARED}

	LIBRARY_NAME              ${NAME}

	LIBRARY_OBJECT_ALIAS_NAME ${FLAME_NAMESPACE}::Streams::Object
	LIBRARY_STATIC_ALIAS_NAME ${FLAME_NAMESPACE}::Streams::Static
	LIBRARY_SHARED_ALIAS_NAME ${FLAME_NAMESPACE}::Streams::Shared

	SOURCE_LIST_FILE          ${SOURCE_LIST_FILE}
	HEADER_LIST_FILE          ${HEADER_LIST_FILE}

	INCLUDE_PATH              ${FLAME_INCLUDE_PATH}

	DEPENDENCY_HEADER_LIST    "${DEPENDENCY_HEADER_LIST}"
	DEPENDENCY_OBJECT_LIST    "${DEPENDENCY_OBJECT_LIST}"
	DEPENDENCY_SHARED_LIST    "${DEPENDENCY_SHARED_LIST};${DEPENDENCY_OBJECT_LIST}"
)
